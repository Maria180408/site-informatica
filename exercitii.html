Exercițiul 1

Rezolvarea unui simplu program în C++ pentru a calcula suma a două numere.


#include <iostream>

int main() {
    int a, b;
    std::cout << "Introduceti doua numere: ";
    std::cin >> a >> b;
    std::cout << "Suma celor doua numere este: " << a + b << std::endl;
    return 0;
}
